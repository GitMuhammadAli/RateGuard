### Two Token Types

| Property            | Access Token              | Refresh Token                     |
| ------------------- | ------------------------- | --------------------------------- |
| **Purpose**         | Authenticate API requests | Get new access tokens             |
| **Lifespan**        | 15 minutes                | 7 days (30 with "Remember Me")    |
| **Stored In**       | Frontend only             | Frontend + Database               |
| **Secret Key**      | `JWT_SECRET`              | `JWT_REFRESH_SECRET` (different!) |
| **Can Be Revoked?** | No (stateless)            | Yes (check database)              |

// In auth.service.ts - refreshToken method

async refreshToken(dto: RefreshTokenDto, ipAddress?: string): Promise<TokenPair> {
  // 1. Verify JWT signature
  const payload = this.jwtService.verify(dto.refreshToken, {
    secret: this.configService.get('jwt.refreshSecret')
  });

  // 2. Hash the incoming token
  const tokenHash = this.hashToken(dto.refreshToken);

  // 3. Find session by hash
  const session = await this.prisma.session.findUnique({
    where: { refreshToken: tokenHash },
    include: { user: true }
  });

  // 4. If not found - token might have been rotated already (STOLEN!)
  if (!session) {
    // ğŸš¨ SECURITY ALERT: Possible token theft!
    this.logger.warn(`Refresh token reuse detected for user ${payload.sub}`);

    // NUCLEAR OPTION: Revoke ALL sessions for this user
    await this.prisma.session.updateMany({
      where: { userId: payload.sub, revokedAt: null },
      data: { revokedAt: new Date() }
    });

    throw new UnauthorizedException('Session invalid - all sessions revoked');
  }

  // 5. If found but already revoked - definitely STOLEN!
  if (session.revokedAt) {
    // ğŸš¨ SECURITY ALERT: Token reuse attempt!
    this.logger.warn(`Attempted use of revoked token for user ${session.userId}`);

    // NUCLEAR OPTION: Revoke ALL sessions
    await this.prisma.session.updateMany({
      where: { userId: session.userId, revokedAt: null },
      data: { revokedAt: new Date() }
    });

    throw new UnauthorizedException('Session revoked - all sessions revoked');
  }

  // 6. Check expiry
  if (new Date() > session.expiresAt) {
    throw new UnauthorizedException('Session expired');
  }

  // 7. Valid - revoke old token, generate new ones
  await this.prisma.session.update({
    where: { id: session.id },
    data: { revokedAt: new Date() }
  });

  // 8. Generate completely new token pair
  return this.generateTokenPair(session.userId, session.user.email, ipAddress);
}

## 8. API Reference

| Method | Endpoint                           | Auth      | Description               |
| ------ | ---------------------------------- | --------- | ------------------------- |
| `POST` | `/api/v1/auth/register`            | Public    | Create new account        |
| `POST` | `/api/v1/auth/login`               | Public    | Login with credentials    |
| `POST` | `/api/v1/auth/refresh`             | Public    | Refresh access token      |
| `POST` | `/api/v1/auth/logout`              | Protected | Revoke current session    |
| `POST` | `/api/v1/auth/logout-all`          | Protected | Revoke all sessions       |
| `GET`  | `/api/v1/auth/verify-email`        | Public    | Verify email with token   |
| `POST` | `/api/v1/auth/resend-verification` | Protected | Resend verification email |
| `POST` | `/api/v1/auth/forgot-password`     | Public    | Request password reset    |
| `POST` | `/api/v1/auth/reset-password`      | Public    | Reset password with token |
| `POST` | `/api/v1/auth/change-password`     | Protected | Change password           |
| `GET`  | `/api/v1/auth/me`                  | Protected | Get current user profile  |
